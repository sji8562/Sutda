package Sutda;

public class Money {
	private int base_money;// 천만원 //나중에 파일로 부터 불러올것임.
	private int bet_money = 20; // 배팅액

	public Money() {

	}

	public int getBase_money() {
		return base_money;
	}

	public int getBet_money() {
		return bet_money;
	}

	public void setBase_money(int base_money) {

		this.base_money = base_money;
	}

	public void setBet_money(int bet_money) {
		this.bet_money = bet_money;
	}

	public void VS(boolean vs) {
		if (vs = true) {
			base_money = (bet_money * 2) + base_money;
		} else {
			base_money = base_money - bet_money;
		}
	}
}
